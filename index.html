    <script>

        const container = document.getElementById('slideshow');

        const totalImages = 30; 

        

        // 1. Automatically generate the 30 image tags pointing to your local 'images' folder

        for (let i = 1; i <= totalImages; i++) {

            let numString = i < 10 ? '0' + i : i; 

            

            let imgElement = document.createElement('img');

            imgElement.src = `images/PORTFOLIO${numString}.jpg`; 

            imgElement.className = 'slide';

            

            if (i === 1) {

                imgElement.classList.add('active');

            }

            

            container.appendChild(imgElement);

        }


        // 2. The Slideshow Logic

        const slides = document.querySelectorAll('.slide');

        let currentSlideIndex = 0;

        const switchInterval = 333; 


        function nextSlide() {

            slides[currentSlideIndex].classList.remove('active');

            currentSlideIndex = (currentSlideIndex + 1) % slides.length;

            slides[currentSlideIndex].classList.add('active');

        }


        setInterval(nextSlide, switchInterval);

    </script>
